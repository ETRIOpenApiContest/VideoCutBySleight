#정상

[HTTP Respone Header]
Access-Control-Allow-Origin:*
Connection:close
Content-Length:0
Content-Type:application/json; charset=UTF-8

[HTTP Respone Body]
{
	"result": [{
			"num": {처리 번호},
			"code": {결과 코드},
			"msg": {결과 메시지},
			"frame": {구분된 프레임 정보},
			"time": {구분된 시간 정보}
	}]
}

#오류

[HTTP Respone Header]
Access-Control-Allow-Origin:*
Connection:close
Content-Length:0
Content-Type:application/json; charset=UTF-8

[HTTP Respone Body]
{
	"request_id": "reserved field",
	"result": -1,
	"reason": {오류 메시지}
}


#분석결과

num       / 장면분할의 처리 번호
code      / 장면분할의 처리결과 코드
msg       / 장면분할의 처리결과 메시지
frame     / 장면분할의 결과로 나온 총 분할시점의 숫자
time      / 분할된 구간의 시작 시점(초)
